// .xx format definition
// char: uint8_t

struct Format
{
	DWORD vertexCount;
	Vertex vertices[vertexCount];
	DWORD lineCount;
	Line lines[lineCount];
	DWORD materialCount;
	Matrial materials[materialCount];
}

struct Vertex
{
	float position[3];
	float normal[3];
	float uv[2];
}

struct Line
{
	WORD vertexA;
	WORD vertexB;
}

struct Material
{
	float diffuse[4];	// r, g, b, a
	float power;
	float specular[3];
	float ambient[3];
	DWORD lineCountPerMaterial;
	char texture[256];	// Shift_JIS string
}
